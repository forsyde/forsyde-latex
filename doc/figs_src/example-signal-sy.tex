\documentclass{standalone}
\usepackage[plot]{forsyde}
\usetikzlibrary{fit}
\begin{document}
\begin{tikzpicture}[]
  \begin{signalsSY}[name=n]
    
    \signalSY{1,2,3,4,5,6,7,8,9}
    \signalSY{2,1,2,3,-2,-6,6}
    \signalSY{9,8,7,6,5,4,3,2,1}
  \end{signalsSY}
  \foreach \i in {1,...,3} {
    \foreach \j in {1,...,7} {
      \node[draw=red, inner sep=0, fit=(n-\i -\j)] {};
      \node[red, rotate=40] at (n-\i -\j) {\tiny n-\i - \j};
    }
  }
  \foreach \i in {1,3} {
    \foreach \j in {8,9} {
      \node[draw=red, inner sep=0, fit=(n-\i -\j)] {};
      \node[red, rotate=40] at (n-\i -\j) {\tiny n-\i - \j};
    }
  } 
\end{tikzpicture}
\end{document}
